  ### Main (then)
```dart

main() async {

  await setColor();

  print('Open Screen');
  print('Load Buttons');
  print('Load Inputs');
  print('Load Images');
}

setColor() { searchData().then( (value) => print('Set Color'));}
searchData() async { await Future.delayed(Duration(seconds: 3), (){ print('Load Data');});}
```
